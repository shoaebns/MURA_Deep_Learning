import pandas as pd
import matplotlib.pyplot as plt

# Load data
train_df = pd.read_csv('/content/drive/MyDrive/Deep_Learning_Project/MURA-v1.1/train_image_paths.csv', header=None, names=['FilePath'])
valid_df = pd.read_csv('/content/drive/MyDrive/Deep_Learning_Project/MURA-v1.1/valid_image_paths.csv', header=None, names=['FilePath'])

# Add Label column to indicate normal (0) or abnormal (1)
for df in [train_df, valid_df]:
    df['Label'] = df.apply(lambda x: 1 if 'positive' in x.FilePath else 0, axis=1)

# Calculate overall counts for training and validation
training_counts = train_df['Label'].value_counts()
validation_counts = valid_df['Label'].value_counts()

# Organize results into a single summary DataFrame
summary_df = pd.DataFrame({
    'Dataset': ['Training', 'Training', 'Validation', 'Validation'],
    'Label': ['Normal (0)', 'Abnormal (1)', 'Normal (0)', 'Abnormal (1)'],
    'Count': [
        training_counts.get(0, 0),  # Get count for Label 0 in training, default to 0 if missing
        training_counts.get(1, 0),  # Get count for Label 1 in training
        validation_counts.get(0, 0),  # Get count for Label 0 in validation
        validation_counts.get(1, 0)  # Get count for Label 1 in validation
    ]
})

# Print the summary
print(summary_df)

# Plot the summary as a bar chart
summary_df.pivot(index='Dataset', columns='Label', values='Count').plot(
    kind='bar', figsize=(10, 6), title='Overall Counts of Normal and Abnormal Cases'
)
plt.xlabel('Dataset')
plt.ylabel('Count')
plt.show()
